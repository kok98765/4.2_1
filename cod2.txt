#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

int main() {
    srand(time(NULL)); 
    int size;
    cout << "Size : ";
    cin >> size;
    int** array = new int* [size];
    for (int i = 0; i < size; i++) {
        array[i] = new int[size];
        for (int j = 0; j < size; j++) {
            array[i][j] = rand() % 10;
           
        }
    }
    for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {

            cout << array[i][j] << " ";

        }
        cout << endl;
    }


    int sum = 0;
    for (int i = 0; i < size; i++) {
        sum += array[i][i];
    }
    cout << "Summa: " << sum << endl;

    for (int i = 0; i < size; i++) {
        delete[] array[i];
    }

    delete[] array;

    return 0;
}
